function Pc(r,h)
{
r_liczba=parseFloat(r)
h_liczba=parseFloat(h)
Pc_wynik=2*Math.PI*r_liczba*r_liczba+2*Math.PI*r_liczba*h_liczba
document.getElementsByName('pole')[0].value=Pc_wynik.toFixed(2);
alert("Pc="+Pc_wynik.toFixed(2)+" cm^2");

}

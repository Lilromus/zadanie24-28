function drabina_kru(D_kru,X_kru)
{
D_kru=parseFloat(D_kru)
X_kru=parseFloat(X_kru)
var drabina=Math.sqrt((D_kru*D_kru) + (X_kru*X_kru));
document.getElementsByName('drabina')[0].value=drabina.toFixed(2);
alert("Długość drabiny="+drabina.toFixed(2)+"m^2");
pokaz1.innerHTML="Długość drabiny="+drabina.toFixed(2)+"m"+"<br>";
}


function kat_kru(D_kru,drabina) {
D_kru=parseFloat(D_kru)
drabina=parseFloat(drabina)
kat=Math.asin(D_kru/drabina)*180/Math.PI;
document.getElementsByName('kat')[0].value=kat.toFixed(2);
alert("Kąt nachylenia wynosi="+kat.toFixed(2)+"°");
pokaz2.innerHTML="Kąt="+kat.toFixed(2)+"°";
}
